    \begin{lstlisting}
    variables (a b c : nat)

    @[simp, to_additive] 
    lemma mul_right_eq_self : a * b = a <-> b = 1 :=
    calc a * b = a ↔ a * b = a * 1 : by rw mul_one
    ...            ↔ b = 1         : mul_left_cancel_iff
    \end{lstlisting}
